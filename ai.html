import librosa
import numpy as np

def pitch_shift(self, waveform: np.array, shift_pitch: float) -> np.array:

  pitch_shift_factor = 2 ** (shift_pitch / 12)
  dummy_sample_rate = 10000

  input_channels = waveform.shape[0]

  if input_channels == 1:
    waveform = np.squeeze(waveform)

  new_waveform = librosa.resample(
    y=waveform,
    orig_sr=dummy_sample_rate,
    target_sr=dummy_sample_rate / pitch_shift_factor,
    res_type='linear',
    axis=-1,
  )

  if input_channels == 1:
    new_waveform = new_waveform[None, :]

  return new_waveform
